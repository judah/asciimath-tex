% Reading an undelimited sequence of letters into a token:
\newtoks\mystuff

\def\mycsname#1{\def\deciderStartB{\deciderStart{#1}}%
\futurelet\tempToken\deciderStartB}
\def\deciderStart#1{\ifcat\tempToken a\def\next{%
\mystuff={#1}\mycsnameloop}%
\else\def\next{#1\loopcs}\fi\next}

\def\mycsnameloop{%
\futurelet\tempToken\decider%
}
\def\decider{%
\ifcat a\tempToken\edef\next##1{\noexpand\mystuff={\the\mystuff ##1}\noexpand\mycsnameloop}%
\else\edef\next{\noexpand\csname\the\mystuff\noexpand\endcsname\noexpand\loopcs}\fi%
\next%
}
